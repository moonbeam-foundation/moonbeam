61impl fc_rpc::EstimateGasAdapter for MoonbeamEGA {
62	fn adapt_request(mut request: TransactionRequest) -> TransactionRequest {
63		// Redirect any call to batch precompile:
64		// force usage of batchAll method for estimation
65		use sp_core::H160;
66		const BATCH_PRECOMPILE_ADDRESS: H160 = H160(hex_literal::hex!(
67			"0000000000000000000000000000000000000808"
68		));
69		const BATCH_PRECOMPILE_BATCH_ALL_SELECTOR: [u8; 4] = hex_literal::hex!("96e292b8");
70		if request.to == Some(BATCH_PRECOMPILE_ADDRESS) {
71			match (&mut request.data.input, &mut request.data.data) {
72				(Some(ref mut input), _) => {
73					if input.0.len() >= 4 {
74						input.0[..4].copy_from_slice(&BATCH_PRECOMPILE_BATCH_ALL_SELECTOR);
75					}
76				}
77				(None, Some(ref mut data)) => {
78					if data.0.len() >= 4 {
79						data.0[..4].copy_from_slice(&BATCH_PRECOMPILE_BATCH_ALL_SELECTOR);
80					}
81				}
82				(_, _) => {}
83			};
84		}
85		request
86	}
87}

153/// Instantiate all Full RPC extensions.
154pub fn create_full<C, P, BE>(
155	deps: FullDeps<C, P, BE>,
156	subscription_task_executor: SubscriptionTaskExecutor,
157	maybe_tracing_config: Option<TracingConfig>,
158	pubsub_notification_sinks: Arc<
159		fc_mapping_sync::EthereumBlockNotificationSinks<
160			fc_mapping_sync::EthereumBlockNotification<Block>,
161		>,
162	>,
163	pending_consenus_data_provider: Box<dyn ConsensusDataProvider<Block>>,
164	para_id: ParaId,
165) -> Result<RpcModule<()>, Box<dyn std::error::Error + Send + Sync>>
166where
167	BE: Backend<Block> + 'static,
168	BE::State: StateBackend<BlakeTwo256>,
169	BE::Blockchain: BlockchainBackend<Block>,
170	C: ProvideRuntimeApi<Block> + StorageProvider<Block, BE> + AuxStore,
171	C: BlockchainEvents<Block>,
172	C: HeaderBackend<Block> + HeaderMetadata<Block, Error = BlockChainError> + 'static,
173	C: CallApiAt<Block>,
174	C: Send + Sync + 'static,
175	C::Api: RuntimeApiCollection,
176	P: TransactionPool<Block = Block, Hash = <Block as BlockT>::Hash> + 'static,
177{
178	use fc_rpc::{
179		Eth, EthApiServer, EthFilter, EthFilterApiServer, EthPubSub, EthPubSubApiServer, Net,
180		NetApiServer, TxPool, TxPoolApiServer, Web3, Web3ApiServer,
181	};
182	use moonbeam_dev_rpc::{DevApiServer, DevRpc};
183	use moonbeam_finality_rpc::{MoonbeamFinality, MoonbeamFinalityApiServer};
184	use moonbeam_rpc_debug::{Debug, DebugServer};
185	use moonbeam_rpc_trace::{Trace, TraceServer};
186	use pallet_transaction_payment_rpc::{TransactionPayment, TransactionPaymentApiServer};
187	use substrate_frame_rpc_system::{System, SystemApiServer};
188
189	let mut io = RpcModule::new(());
190	let FullDeps {
191		client,
192		pool,
193		graph,
194		is_authority,
195		network,
196		sync,
197		filter_pool,
198		ethapi_cmd,
199		command_sink,
200		frontier_backend,
201		backend: _,
202		max_past_logs,
203		max_block_range,
204		fee_history_limit,
205		fee_history_cache,
206		dev_rpc_data,
207		overrides,
208		block_data_cache,
209		forced_parent_hashes,
210	} = deps;
211
212	io.merge(System::new(Arc::clone(&client), Arc::clone(&pool)).into_rpc())?;
213	io.merge(TransactionPayment::new(Arc::clone(&client)).into_rpc())?;
214
215	// TODO: are we supporting signing?
216	let signers = Vec::new();
217
218	enum Never {}
219	impl<T> fp_rpc::ConvertTransaction<T> for Never {
220		fn convert_transaction(&self, _transaction: pallet_ethereum::Transaction) -> T {
221			// The Never type is not instantiable, but this method requires the type to be
222			// instantiated to be called (`&self` parameter), so if the code compiles we have the
223			// guarantee that this function will never be called.
224			unreachable!()
225		}
226	}
227	let convert_transaction: Option<Never> = None;
228
229	// Need to clone it to avoid moving of `client` variable in closure below.
230	let client_for_cidp = client.clone();
231
232	let pending_create_inherent_data_providers = move |block, _| {
233		// Use timestamp in the future
234		let timestamp = sp_timestamp::InherentDataProvider::new(
235			Timestamp::current()
236				.saturating_add(RELAY_CHAIN_SLOT_DURATION_MILLIS.saturating_mul(100))
237				.into(),
238		);
239
240		let maybe_current_para_head = client_for_cidp.expect_header(block);
241		async move {
242			let current_para_block_head = Some(polkadot_primitives::HeadData(
243				maybe_current_para_head?.encode(),
244			));
245
246			let builder = RelayStateSproofBuilder {
247				para_id,
248				// Use a future relay slot (We derive one from the timestamp)
249				current_slot: polkadot_primitives::Slot::from(
250					timestamp
251						.timestamp()
252						.as_millis()
253						.saturating_div(RELAY_CHAIN_SLOT_DURATION_MILLIS),
254				),
255				included_para_head: current_para_block_head,
256				..Default::default()
257			};
258
259			// Create a dummy parachain inherent data provider which is required to pass
260			// the checks by the para chain system. We use dummy values because in the 'pending context'
261			// neither do we have access to the real values nor do we need them.
262			let (relay_parent_storage_root, relay_chain_state) =
263				builder.into_state_root_and_proof();
264
265			let vfp = PersistedValidationData {
266				// This is a hack to make `cumulus_pallet_parachain_system::RelayNumberStrictlyIncreases`
267				// happy. Relay parent number can't be bigger than u32::MAX.
268				relay_parent_number: u32::MAX,
269				relay_parent_storage_root,
270				..Default::default()
271			};
272			let parachain_inherent_data = ParachainInherentData {
273				validation_data: vfp,
274				relay_chain_state,
275				downward_messages: Default::default(),
276				horizontal_messages: Default::default(),
277				relay_parent_descendants: Default::default(),
278				collator_peer_id: None,
279			};
280
281			Ok((timestamp, parachain_inherent_data))
282		}
283	};
284
285	io.merge(
286		Eth::<_, _, _, _, _, _, MoonbeamEthConfig<_, _>>::new(
287			Arc::clone(&client.clone()),
288			Arc::clone(&pool),
289			graph.clone(),
290			convert_transaction,
291			Arc::clone(&sync),
292			signers,
293			Arc::clone(&overrides),
294			Arc::clone(&frontier_backend),
295			is_authority,
296			Arc::clone(&block_data_cache),
297			fee_history_cache,
298			fee_history_limit,
299			10,
300			true,
301			forced_parent_hashes,
302			pending_create_inherent_data_providers,
303			Some(pending_consenus_data_provider),
304		)
305		.replace_config::<MoonbeamEthConfig<C, BE>>()
306		.into_rpc(),
307	)?;
308
309	if let Some(filter_pool) = filter_pool {
310		io.merge(
311			EthFilter::new(
312				client.clone(),
313				frontier_backend.clone(),
314				graph.clone(),
315				filter_pool,
316				500_usize, // max stored filters
317				max_past_logs,
318				max_block_range,
319				block_data_cache,
320			)
321			.into_rpc(),
322		)?;
323	}
324
325	io.merge(
326		Net::new(
327			Arc::clone(&client),
328			network.clone(),
329			// Whether to format the `peer_count` response as Hex (default) or not.
330			true,
331		)
332		.into_rpc(),
333	)?;
334
335	io.merge(Web3::new(Arc::clone(&client)).into_rpc())?;
336	io.merge(
337		EthPubSub::new(
338			pool,
339			Arc::clone(&client),
340			sync.clone(),
341			subscription_task_executor,
342			overrides,
343			pubsub_notification_sinks.clone(),
344		)
345		.into_rpc(),
346	)?;
347
348	if ethapi_cmd.contains(&EthApiCmd::Txpool) {
349		io.merge(TxPool::new(Arc::clone(&client), graph).into_rpc())?;
350	}
351
352	io.merge(MoonbeamFinality::new(client.clone(), frontier_backend.clone()).into_rpc())?;
353
354	if let Some(command_sink) = command_sink {
355		io.merge(
356			// We provide the rpc handler with the sending end of the channel to allow the rpc
357			// send EngineCommands to the background block authorship task.
358			ManualSeal::new(command_sink).into_rpc(),
359		)?;
360	};
361
362	if let Some((downward_message_channel, hrmp_message_channel, additional_relay_offset)) =
363		dev_rpc_data
364	{
365		io.merge(
366			DevRpc {
367				downward_message_channel,
368				hrmp_message_channel,
369				additional_relay_offset,
370			}
371			.into_rpc(),
372		)?;
373	}
374
375	if let Some(tracing_config) = maybe_tracing_config {
376		if let Some(trace_filter_requester) = tracing_config.tracing_requesters.trace {
377			io.merge(
378				Trace::new(
379					client,
380					trace_filter_requester,
381					tracing_config.trace_filter_max_count,
382					tracing_config.max_block_range,
383				)
384				.into_rpc(),
385			)?;
386		}
387
388		if let Some(debug_requester) = tracing_config.tracing_requesters.debug {
389			io.merge(Debug::new(debug_requester).into_rpc())?;
390		}
391	}
392
393	Ok(io)
394}

407/// Spawn the tasks that are required to run Moonbeam.
408pub fn spawn_essential_tasks<B, C, BE>(
409	params: SpawnTasksParams<B, C, BE>,
410	sync: Arc<SyncingService<B>>,
411	pubsub_notification_sinks: Arc<
412		fc_mapping_sync::EthereumBlockNotificationSinks<
413			fc_mapping_sync::EthereumBlockNotification<B>,
414		>,
415	>,
416) where
417	C: ProvideRuntimeApi<B> + BlockOf,
418	C: HeaderBackend<B> + HeaderMetadata<B, Error = BlockChainError> + 'static,
419	C: BlockchainEvents<B> + StorageProvider<B, BE>,
420	C: Send + Sync + 'static,
421	C::Api: EthereumRuntimeRPCApi<B>,
422	C::Api: BlockBuilder<B>,
423	B: BlockT<Hash = H256> + Send + Sync + 'static,
424	B::Header: HeaderT<Number = u32>,
425	BE: Backend<B> + 'static,
426	BE::State: StateBackend<BlakeTwo256>,
427{
428	// Frontier offchain DB task. Essential.
429	// Maps emulated ethereum data to substrate native data.
430	match *params.frontier_backend {
431		fc_db::Backend::KeyValue(ref b) => {
432			params.task_manager.spawn_essential_handle().spawn(
433				"frontier-mapping-sync-worker",
434				Some("frontier"),
435				MappingSyncWorker::new(
436					params.client.import_notification_stream(),
437					Duration::new(6, 0),
438					params.client.clone(),
439					params.substrate_backend.clone(),
440					params.overrides.clone(),
441					b.clone(),
442					3,
443					0,
444					SyncStrategy::Parachain,
445					sync.clone(),
446					pubsub_notification_sinks.clone(),
447				)
448				.for_each(|()| futures::future::ready(())),
449			);
450		}
451		fc_db::Backend::Sql(ref b) => {
452			params.task_manager.spawn_essential_handle().spawn_blocking(
453				"frontier-mapping-sync-worker",
454				Some("frontier"),
455				fc_mapping_sync::sql::SyncWorker::run(
456					params.client.clone(),
457					params.substrate_backend.clone(),
458					b.clone(),
459					params.client.import_notification_stream(),
460					fc_mapping_sync::sql::SyncWorkerConfig {
461						read_notification_timeout: Duration::from_secs(10),
462						check_indexed_blocks_interval: Duration::from_secs(60),
463					},
464					fc_mapping_sync::SyncStrategy::Parachain,
465					sync.clone(),
466					pubsub_notification_sinks.clone(),
467				),
468			);
469		}
470	}
471
472	// Frontier `EthFilterApi` maintenance.
473	// Manages the pool of user-created Filters.
474	if let Some(filter_pool) = params.filter_pool {
475		// Each filter is allowed to stay in the pool for 100 blocks.
476		const FILTER_RETAIN_THRESHOLD: u64 = 100;
477		params.task_manager.spawn_essential_handle().spawn(
478			"frontier-filter-pool",
479			Some("frontier"),
480			EthTask::filter_pool_task(
481				Arc::clone(&params.client),
482				filter_pool,
483				FILTER_RETAIN_THRESHOLD,
484			),
485		);
486	}
487
488	// Spawn Frontier FeeHistory cache maintenance task.
489	params.task_manager.spawn_essential_handle().spawn(
490		"frontier-fee-history",
491		Some("frontier"),
492		EthTask::fee_history_task(
493			Arc::clone(&params.client),
494			Arc::clone(&params.overrides),
495			params.fee_history_cache,
496			params.fee_history_limit,
497		),
498	);
